SCRIPT USED TO DETERMINE GABOR-DETECTION PSYCHOMETRIC FUNCTION AT SEVERALSPATIAL FREQUENCIES ( FROM WHICH THRESHOLDS WERE THEN PLOTTED ):	global Frequency   % used by stimulus-generating routines	% try several wavelengths in turn	for Lambda=4:2:24,		Frequency=1/Lambda;		unix('rm CSF*.mat'); % remove old stimulus files		% get psychometric function ( 200 trials per level )		[ Trials Alpha Beta ]=RunSDS('CSF0','CSF1', ...			[ 0 1 2 4 8 16 32 64 100 ],200);		% save all trials and resulting fit		eval([ 'save icpm_' num2str(Lambda) ' Trials Alpha Beta' ]);	endIMAGE AND PROBABILITY FUNCTIONS:  STIMULUS ABSENT	function [ Img ] = CSF0Img( StimLev, Unc )		Size=64;		Img = 5*ones(Size,Size);		return	function [ P ] = CSF0Prob( Unc )		if Unc==-1,			P = [ 1 ];		elseif Unc(1)==1,			P=1;		else			P=0;		endIMAGE AND PROBABILITY FUNCTIONS:  STIMULUS PRESENT	function [ Img ] = CSF1Img( StimLev, Unc )		global Frequency;		Size=64;		Img = 5*( (StimLev/100)*(Sine2D(Size,1/Frequency,0,0).* ...			Gauss2D(Size,0.5/Frequency)) + 1 );	function [ P ] = CSF1Prob( Unc )		if Unc==-1,			P = [ 1 ];		elseif Unc(1)==1,			P=1;		else			P=0;		end
